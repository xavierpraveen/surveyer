    **supabase/migrations/20260315000003_views.sql**

    Create SECURITY DEFINER analytics views/functions. These are the ONLY way analytics data should be accessed — direct table SELECT on responses/response_answers is blocked for most roles.

    ```sql
    -- View: dimension scores per survey (respects privacy threshold from app_settings)
    CREATE OR REPLACE VIEW public.v_dimension_scores
    WITH (security_invoker = false)  -- SECURITY DEFINER behavior
    AS
    SELECT
      dm.survey_id,
      d.id as dimension_id,
      d.name as dimension_name,
      d.slug as dimension_slug,
      COUNT(DISTINCT r.id) as respondent_count,
      ROUND(AVG(ra.numeric_value), 2) as avg_score,
      ROUND(100.0 * SUM(CASE WHEN ra.numeric_value >= 4 THEN 1 ELSE 0 END) / NULLIF(COUNT(ra.id), 0), 1) as favorable_pct,
      ROUND(100.0 * SUM(CASE WHEN ra.numeric_value = 3 THEN 1 ELSE 0 END) / NULLIF(COUNT(ra.id), 0), 1) as neutral_pct,
      ROUND(100.0 * SUM(CASE WHEN ra.numeric_value <= 2 THEN 1 ELSE 0 END) / NULLIF(COUNT(ra.id), 0), 1) as unfavorable_pct
    FROM derived_metrics dm
    JOIN dimensions d ON d.id = dm.dimension_id
    WHERE dm.segment_type = 'overall'
    GROUP BY dm.survey_id, d.id, d.name, d.slug;

    -- View: participation rates per survey (uses participation_tokens, not responses)
    CREATE OR REPLACE VIEW public.v_participation_rates AS
    SELECT
      pt.survey_id,
      COUNT(DISTINCT pt.user_id) as token_count,
      pt.department_id,
      d.name as department_name
    FROM participation_tokens pt
    LEFT JOIN departments d ON d.id = pt.department_id
    GROUP BY pt.survey_id, pt.department_id, d.name;

    -- View: public action items with latest update
    CREATE OR REPLACE VIEW public.v_public_actions AS
    SELECT
      ai.id,
      ai.title,
      ai.problem_statement,
      ai.status,
      ai.priority,
      ai.target_date,
      ai.success_criteria,
      ai.department_id,
      d.name as department_name,
      ai.survey_id,
      ai.created_at
    FROM action_items ai
    LEFT JOIN departments d ON d.id = ai.department_id
    WHERE ai.is_public = TRUE;

    -- RPC: get_dimension_scores_for_survey (enforces privacy threshold)
    CREATE OR REPLACE FUNCTION public.get_dimension_scores_for_survey(
      p_survey_id UUID,
      p_min_respondents INT DEFAULT 5
    )
    RETURNS TABLE (
      dimension_id UUID,
      dimension_name TEXT,
      dimension_slug TEXT,
      avg_score NUMERIC,
      favorable_pct NUMERIC,
      neutral_pct NUMERIC,
      unfavorable_pct NUMERIC,
      respondent_count BIGINT,
      below_threshold BOOLEAN
    )
    LANGUAGE plpgsql
    SECURITY DEFINER
    SET search_path = public
    AS $$
    BEGIN
      RETURN QUERY
      SELECT
        vds.dimension_id,
        vds.dimension_name,
    ```
