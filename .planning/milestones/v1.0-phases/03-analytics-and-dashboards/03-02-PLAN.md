// Dashboard filter state (matches nuqs URL params)
export interface DashboardFilters {
  surveyId: string | null
  department: string | null
  role: string | null
  tenureBand: string | null
}

// Per-dimension score row (from get_dimension_scores_for_survey RPC)
export interface DimensionScore {
  dimensionId: string
  dimensionName: string
  dimensionSlug: string
  avgScore: number | null       // null when below_threshold = true
  favorablePct: number | null
  neutralPct: number | null
  unfavorablePct: number | null
  respondentCount: number
  belowThreshold: boolean
}

// Overall org KPIs shown in the KPI strip
export interface OrgKpis {
  overallHealthScore: number | null   // avg of all dimension avg_scores
  participationRate: number           // 0–100 percentage
  totalResponses: number
  surveyPeriod: string                // e.g. "Q1 2026"
  dimensionsBelowThreshold: number
  surveyId: string
  surveyTitle: string
}

// One row in the department heatmap (one per department)
export interface HeatmapRow {
  department: string
  scores: {
    dimensionId: string
    dimensionName: string
    avgScore: number | null
    belowThreshold: boolean
  }[]
}

// One data point in the trend chart
export interface TrendPoint {
  surveyId: string
  surveyTitle: string
  closedAt: string               // ISO date string
  dimensionId: string
  dimensionName: string
  avgScore: number | null
  belowThreshold: boolean
}

// Participation broken down by department
export interface ParticipationBreakdown {
  department: string
  departmentId: string | null
  respondentCount: number
}

// Qualitative theme (from qualitative_themes table)
export interface QualitativeTheme {
  id: string
  theme: string
  tagCluster: string[]
  summary: string | null
  isPositive: boolean            // true = suggestion, false = issue
  tagCount: number               // tag_cluster.length as proxy for frequency
}

// Action item for public display (from v_public_actions view)
export interface PublicAction {
  id: string
  title: string
  problemStatement: string | null
  status: 'identified' | 'planned' | 'in_progress' | 'blocked' | 'completed'
  priority: string | null
  targetDate: string | null
  successCriteria: string | null
  departmentName: string | null
}

// Full payload returned to leadership dashboard RSC
export interface LeadershipDashboardData {
  kpis: OrgKpis
  dimensionScores: DimensionScore[]
  heatmapRows: HeatmapRow[]
  trendPoints: TrendPoint[]
  participationBreakdown: ParticipationBreakdown[]
  qualitativeThemes: QualitativeTheme[]
  publicActions: PublicAction[]           // read-only preview of public action items (is_public=true)
  availableSurveys: { id: string; title: string; closedAt: string }[]
}

// Manager dashboard payload
export interface ManagerDashboardData {
  teamName: string | null
  participationCount: number     // always visible
  totalTeamMembers: number       // denominator for participation %
  dimensionScores: DimensionScore[] | null  // null if team < 5 respondents
  belowThreshold: boolean
  surveyTitle: string | null
}

// Public /results page payload
export interface PublicResultsData {
  hasData: boolean               // false = no closed+computed survey yet
  surveyTitle: string | null
  surveyClosedAt: string | null
  kpis: Pick<OrgKpis, 'overallHealthScore' | 'participationRate' | 'totalResponses'> | null
  dimensionScores: DimensionScore[]
  qualitativeThemes: QualitativeTheme[]
  publicActions: PublicAction[]
}

**computeDerivedMetrics(surveyId: string)**
- Auth check: getUser() — must be admin, leadership, or survey_analyst role (check user.app_metadata.role)
- `const db = supabaseAdmin as any`
- `const { data, error } = await db.rpc('compute_derived_metrics', { p_survey_id: surveyId })`
- Return `{ success: true, data: { rowsInserted: data } }` or `{ success: false, error: error.message }`

**getLeadershipDashboardData(filters: DashboardFilters)**
- Uses createSupabaseServerClient() (authenticated reads via RLS-allowed views)
- Determine target surveyId: if filters.surveyId provided use it; else fetch most recent closed survey that has rows in derived_metrics
- Fetch dimension scores via `supabase.rpc('get_dimension_scores_for_survey', { p_survey_id: surveyId, p_min_respondents: 5 })`
- Fetch heatmap: `(supabaseAdmin as any).from('derived_metrics').select('*').eq('survey_id', surveyId).eq('segment_type', 'department')` grouped by segment_value × dimension
  - Apply filters: if filters.department, only return heatmap rows for that department
  - Threshold: heatmap cell score = null if respondent_count < 5; belowThreshold = true
- Fetch trend: for the last 5 closed surveys, fetch derived_metrics where segment_type='overall'
- Fetch participation: from v_participation_rates view, sum token counts per department for the survey
- Fetch qualitative themes: from qualitative_themes table where survey_id = target survey
- Fetch public action items: from v_public_actions view — `(supabaseAdmin as any).from('v_public_actions').select('*').eq('survey_id', surveyId)` — map to PublicAction[]. If the view does not filter by survey_id (action items may have no survey link), also include rows where survey_id IS NULL. Map DB snake_case to camelCase (problem_statement → problemStatement, target_date → targetDate, etc.)
- Fetch available surveys: all closed surveys with at least one derived_metrics row
- Assemble and return LeadershipDashboardData (including publicActions field)

**getManagerDashboardData()**
- Auth check: getUser() must be role=manager
- Find manager's profile_id: query profiles where user_id = user.id
- Find team: query teams where manager_id = manager's profile_id
- Find team members: query profiles where team_id = team.id
- Count team members (totalTeamMembers)
- Find most recent closed+computed survey_id
- Count participation tokens for team members in that survey (by cross-referencing participation_tokens with user_ids of team members — safe: participation_tokens.user_id is visible to manager via service role)
  - Use supabaseAdmin to count participation_tokens where survey_id = target and user_id IN (team member user_ids)
- If participationCount < 5: return with dimensionScores=null, belowThreshold=true
- Else: fetch derived_metrics for team segment — use segment_type='overall' as proxy (manager sees org-wide for now; team-level segment requires future work); belowThreshold=false if participationCount >= 5
- Return ManagerDashboardData

**getPublicResultsData()**
- No auth role restriction — any authenticated user can call this
- Fetch most recent closed survey with computed derived_metrics rows
- If none: return { hasData: false, ... } with all nulls
- Fetch dimension scores via get_dimension_scores_for_survey RPC (company-wide only, no segment filter)
- Fetch qualitative themes from qualitative_themes table
- Fetch public action items from v_public_actions view
- Return PublicResultsData
